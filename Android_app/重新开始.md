def trans(vars_list):
    new_var_list = []
    var_dict = {}
    for var in vars_list:
        global_vars = globals()
        for k, v in global_vars.items():
            var_dict[id(v)] = k
        new_var_list.append(var_dict[id(var)])
    return new_var_list
